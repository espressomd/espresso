  PYTHON_SCRIPTS_TEST(${ARGV} TYPE sample)  # forward arguments
  set(samples_tests ${samples_tests} ${TEST_FILE_CONFIGURED} PARENT_SCOPE)
endfunction(SAMPLE_TEST)
